func isPalindrome(x int) bool {
    b:=[]int{}
    c:=0
    var t bool=true
    if x<0{
       t =false
    }
    for x>0{
        c=x%10
        b=append(b,c)
        x=x/10
    }
    var n=len(b)
    var i=0
    var j=n-1
    for j-i>=1{
        if b[i]==b[j]{
            i++
            j--
        }else{
           t=false
            break
        }
    }
    
       return t
    }
